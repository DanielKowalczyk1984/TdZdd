122  NodeTableHandler<S::ARITY> diagram;
123  InstantDdBuilder<S,true> idb(entity(), diagram, os, cut);
124  NodeId root;
125  idb.initialize(root);
126 
127  os << "digraph \"" << title << "\" {\n";

135  }
136 
137  if (root == 1) {
138  os << " \"^\" [shape=none,label=\"" << title << "\"];\n";
139  os << " \"^\" -> \"" << root << "\" [style=dashed";
140  if (root.getAttr()) os << ",arrowtail=dot";
141  os << "];\n";
142  if (cut) os << " \"" << root
143  << "\" [shape=square,fixedsize=true,width=0.2,label=\"\"];\n";
144  }
145  else if (!cut && root != 0) {
146  os << " \"^\" [shape=none,label=\"" << title << "\"];\n";
147  os << " \"^\" -> \"" << root << "\" [style=dashed";
148  if (root.getAttr()) os << ",arrowtail=dot";
149  os << "];\n";
150  }
151  else if (!title.empty()) {
152  os << " labelloc=\"t\";\n";
153  os << " label=\"" << title << "\";\n";
154  }
155 
156  for (int i = root.row(); i >= 1; --i) {
157  idb.construct(i);
158  diagram.derefLevel(i);
159  }
160 
161  if (!cut && root != 0) {
162  os << " \"" << NodeId(1) << "\" ";
163  if (cut) {
164  os << "[shape=square,fixedsize=true,width=0.2,label=\"\"];\n";
165  }
166  else {
167  os << "[shape=square,label=\"⊤\"];\n";
168  }
169  }
